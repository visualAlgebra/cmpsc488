Account:

{
"accountID": string,
"bio": string (limited 1000 chars?),
"creations": Array of strings (lesson/problem IDs),
"timeCreated": Timestamp, 
}



Lesson: 

{
"creatorAccountID": string,
"description": string (limited 1000 chars?),
"creations": array of problems/lessons,
"timeCreated": timestamp,
}



Problem:

{
"creatorAccountID": string,
"description": string (limited 1000 chars?),
"attachedLesson": string ("null" if no lesson),
"startExpression": array of numbers,
"goalExpression": array of numbers,
"timeCreated": timestamp
}
